# 1. 暴力修改网络配置文件，彻底断开与启动服务的冲突
sed -i 's/192.168.1.1/192.168.51.1/g' /etc/config/network

# 2. 修改系统主机名
uci set system.@system[0].hostname='Ecom-Gateway'
uci commit system

# 3. 动态配置并强制开启所有 Wi-Fi 接口
uci show wireless | grep =wifi-iface | cut -d'.' -f2 | cut -d'=' -f1 | while read -r iface; do
    uci set wireless.${iface}.ssid='Ecom-WiFi'
    uci set wireless.${iface}.encryption='psk2'
    uci set wireless.${iface}.key='password'
    uci set wireless.${iface}.disabled='0'
done

# 强制唤醒可能被禁用的无线物理射频
[ -e /etc/config/wireless ] && {
    uci set wireless.radio0.disabled='0'
    uci set wireless.radio1.disabled='0'
    uci commit wireless
}

# 4. 【核心增加】将默认 root 登录密码强行修改为: password
# 使用 Linux 标准的 shadow 密码哈希生成器，100% 兼容 OpenWrt 的 musl-libc
echo "password" | passwd root --stdin 2>/dev/null || echo -e "password\npassword" | passwd root

# 5. 安全重启网络，让 IP 和 Wi-Fi 彻底平滑就位
/etc/init.d/system restart
/etc/init.d/network restart
wifi reload
